                                                                                  # 介绍

                                                                                  ElasticSearch 是一个基于Lucene的搜索服务器

                                                                                  是一个分布式、高扩展、高实时的搜索与数据分析引警

                                                                                  基于RESTful web接口

                                                                                  Elasticsearch 是用Java语言开发的，并作为Apache许可条款下的开放源码发布，是一种流行的企业级搜索引警

                                                                                  MySQL有事务性而ElasticSearch没有事务性,所以你删了的数据是无法恢复的。

                                                                                  ElasticSearch没有物理外键这个特性，,如果你的数据强一致性要求比较高还是建议慎用

                                                                                  ElasticSearch和MySql分工不同，MySQL责存储数据，ElasticSearch负责搜索数据

                                                                                  # 倒排索引

                                                                                  将各个文档中的内容，进行分词，形成词条。然后记录词条和数据的唯一标识（id）的对应关系，形成的索引。

                                                                                  形成倒排索引时候，词条会进行排序，来解决词条数据大查询效率的问题。

                                                                                  # 核心概念

                                                                                  索引 (index)

                                                                                  ElasticSearch存储数据的地方，可以理解成关系型数据库中的数据库概念

                                                                                  映射 (mapping)

                                                                                  mapping定义了每个字段的类型、字段所使用的分词器等、相当于关系型数据库中的表结构文档

                                                                                  (document) Elasticsearch中的最小数据单元，常以ison格式显示。一个document相当于关系型数据库中的一行数据.

                                                                                  倒排索引

                                                                                  一个倒排索引由文档中所有不重复词的列表构成，对于其中每个词，对应一个包含它的文档id列表

                                                                                  类型 (type)

                                                                                  type就像 类表。如用户表、角色表等。在Elasticsearch7.X默认type为 doc

                                                                                  ES 5.x中一个index可以有多种type。

                                                                                  ES 6.x中一个index只能有一种type。

                                                                                  ES 7.x以后，将逐步移除type这个概念，现在的操作已经不再使用，默认 doc

                                                                                  # 创建索引

                                                                                   public static void createIndex() throws IOException {

    RestHighLevelClient client = new RestHighLevelClient(
        RestClient.builder(
            new HttpHost("127.0.0.1", 9200, "http")));

    IndicesClient indices = client.indices();

    CreateIndexRequest persons = new CreateIndexRequest("persons");
    // 添加mappings
    String mapping="{\n"
        + "      \"properties\" : {\n"
        + "        \"address\" : {\n"
        + "          \"type\" : \"text\",\n"
        + "          \"analyzer\" : \"ik_max_word\"\n"
        + "        },\n"
        + "        \"age\" : {\n"
        + "          \"type\" : \"integer\"\n"
        + "        },\n"
        + "        \"name\" : {\n"
        + "          \"type\" : \"keyword\"\n"
        + "        }\n"
        + "      }\n"
        + "    }";
    persons.mapping(mapping, XContentType.JSON);

    CreateIndexResponse createIndexResponse =
        indices.create(persons, RequestOptions.DEFAULT);

    System.out
        .println("createIndexResponse.isAcknowledged() = " + createIndexResponse.isAcknowledged());

  }

                                                                                  # 获取索引

                                                                                  public static void getIndex() throws IOException {
    RestHighLevelClient client = new RestHighLevelClient(
        RestClient.builder(
            new HttpHost("127.0.0.1", 9200, "http")));

    IndicesClient indices = client.indices();

    GetIndexRequest request = new GetIndexRequest("persons");

    GetIndexResponse getIndexResponse =
        indices.get(request, RequestOptions.DEFAULT);

    Map<String, MappingMetadata> mappings =
        getIndexResponse.getMappings();

    Set<Entry<String, MappingMetadata>> entries = mappings.entrySet();

    entries.forEach(entry->{

      String key = entry.getKey();
      System.out.println("key = " + key);

      MappingMetadata value = entry.getValue();
      System.out.println("value = " + value.getSourceAsMap());
    });

  }

                                                                                  # 删除索引

                                                                                    public static void deleteIndex() throws IOException {
    RestHighLevelClient client = new RestHighLevelClient(
        RestClient.builder(
            new HttpHost("127.0.0.1", 9200, "http")));

    IndicesClient indices = client.indices();

    DeleteIndexRequest persons = new DeleteIndexRequest("persons");

    AcknowledgedResponse delete = indices.delete(persons, RequestOptions.DEFAULT);

    System.out.println("delete = " + delete.isAcknowledged());

  }

                                                                                  # 判断索引是否存在

                                                                                  public static void exist() throws IOException {
    RestHighLevelClient client = new RestHighLevelClient(
        RestClient.builder(
            new HttpHost("127.0.0.1", 9200, "http")));

    IndicesClient indices = client.indices();

    GetIndexRequest persons = new GetIndexRequest("persons");
    boolean exists = indices.exists(persons, RequestOptions.DEFAULT);

    System.out.println("exists = " + exists);
  }

                                                                                  # 添加文档

                                                                                  public static void addDoc() throws IOException {
    RestHighLevelClient client = new RestHighLevelClient(
        RestClient.builder(
            new HttpHost("127.0.0.1", 9200, "http")));

    HashMap<String, Object> objectObjectHashMap = new HashMap<>();
    objectObjectHashMap.put("name","张三");
    objectObjectHashMap.put("address","北京昌平");
    objectObjectHashMap.put("age",30);
    IndexRequest indexRequest = new IndexRequest("persons").id("1").source(objectObjectHashMap);

    IndexResponse index = client.index(indexRequest, RequestOptions.DEFAULT);

    System.out.println("index = " + index.getId());
  }


  public static void addDoc2() throws IOException {
