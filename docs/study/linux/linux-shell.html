echo $1 $2

                                                              # shell 常用变量

                                                              $?  获取上一个命令执行的结果 0 成功 非0 不成功   常用

$$ 获取当前脚本执行的进程pid

$_ 获取上一指令的最后一个参数

$! 获取上一个在后台运行的进程id  pid

[root@w1 myShell]# echo hello
hello

1
[root@w1 myShell]#

                                                              # bash shell 内置命令

                                                              echo 

-n 不换行
-e 遇到转义字符进行转义

转义字符如下：
\n  换行
\r  回车

eval args  当程序执行到eval 时，shell 读入参数args 并将他们组成一个新的命令，然后执行
例子：
[root@w1 myShell]# cat noeval.sh 
#!/bin/bash

echo \$$#

[root@w1 myShell]# sh noeval.sh a b c
$3

# eval
[root@w1 myShell]# cat eval.sh 
#1/bin/bash

eval  "echo \$$#"

[root@w1 myShell]# sh eval.sh a b
b

5

                                                              # 变量的数值计算

                                                              # (())  数值计算

[root@w1 ~]# echo $((100*3))
300
[root@w1 ~]# ((a=100*3))
[root@w1 ~]# echo $a
300

# false 0  true 1
[root@w1 ~]# echo $((1<0))
0
[root@w1 ~]# echo $((1>0))
1

# let 变量运算 相当于 ((表达式)) 但是 (()) 效率最高

[root@w1 ~]# let a=8*3
[root@w1 ~]# echo $a
24
[root@w1 ~]# ((a=8*3))
[root@w1 ~]# echo $a
24

# expr 变量求值  计算符两边要有空格

[root@w1 ~]# expr 2 + 2
4
[root@w1 ~]# expr 2+2
2+2

[root@w1 ~]# i=3
[root@w1 ~]# a=`expr $i + 3`
[root@w1 ~]# echo $a
6

#  &>/dev/null 表示不保留任何输出
[root@w1 ~]# i=5
[root@w1 ~]# expr $i + 6 &>/dev/null
[root@w1 ~]# expr $i + 6
11

[root@w1 ~]# cat expr.sh 


for n in  i am a oldboy linux welcome to our training
do
  if [ `expr length $n` -le 3 ] 
  then 
   echo $n
  fi
done

[root@w1 ~]# sh expr.sh 
i
am
a
to
our

                                                              # 基于shell 传参 read

                                                               语法： 
 read [参数][变量名]
 常用参数：
 -p prompt:设置提示信息
 -t timeout: 输入的等待时间，单位默认为秒
 
 [root@w1 ~]# read -t 10 -p "请输入 num:" num
请输入 num:100
[root@w1 ~]# echo $num
100

 [root@w1 ~]# read -t 10 -p "请输入 a1 a2:" a1 a2
请输入 a1 a2:1 2
[root@w1 ~]# echo $a1  $a2
1 2

# -d 是否是目录   -e 文件是否存在（不区分是文件还是目录）

[root@w1 ~]# test -z "" && echo true || echo false
true

# [<表达式>]

[root@w1 ~]# [ -f test.log ] && echo true || echo false
false

# [[<表达式>]]

[root@w1 ~]# [[ -f test.log ]] || echo false
false

## 使用bash 命令参数调试
# -n 不会执行该脚本 仅检查语法是否有问题 并给出提示
# -v 在执行脚本时，先脚本的内容输出到屏幕上，然后执行脚本，如有错误，也会给出提示
# -x 将执行的脚本内容及输出显示到屏幕上 这是对调试很有用的参数
[root@w1 ~]# sh -nvx for.sh 
#!/bin/bash

for n in a b c d
do
   echo $n
done
