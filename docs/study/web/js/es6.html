                                      # let 和 const

                                      //   var 是变量提升：自动向外定义

 //  var a = 10;

 // let 不能重复定义

 // let.1 提示在变量实例化之前不能使用
 // console.log(b);
 // let b = 10;

 // let.2 块级变量

 // if(1===1){

 // }

 // // 报 c is not defined
 // console.log(c);

 //  for(let i = 0;i < 10 ; i ++){

 //  }

 // person.age = 30

                                      # 继承

                                        //  继承
  class Animal {

    constructor(name, age) {
      this.name = name
      this.age = age
    }
    sayName() {
      console.log(this.name);
    }
    sayAge() {
      console.log(this.age);
    }
  }

  class Dog extends Animal{

   constructor(name,age,color){
    super(name,age)
    this.color = color
   }
   sayColor(){
     console.log(this.color);
   }
  }

  const dog = new Dog('小黄',20,'黄色')
  console.log(dog);
