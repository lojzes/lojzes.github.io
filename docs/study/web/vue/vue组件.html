     `
  })
  // 然后挂载
  app.mount("#app");

                                    # 子组件通过 $emit 事件， 告诉父组件修改行为

                                     var app = Vue.createApp({
    data() {
      return {
          obj : 1
      }
    },
    methods: {
      handleAddOne(){
          this.obj += 1
      }
    },
    template: `
          <hello-world  :msg="obj" @add-one="handleAddOne"/>
      `,
  })

  // String number Boolean Function 
  app.component('hello-world', {
     props: ['msg'],
     methods: {
      handleClick(){
       this.$emit('addOne')
      }
     },
    template: `
       <div @click="handleClick">{{msg}}</div>
    `
  })

  // 传参
  this.$emit('addOne',10)

// 使用 emits 当注释，表示当前子组件要向外发送的事件
 // String number Boolean Function 
  app.component('hello-world', {
     props: ['msg'],
     emits:['addOne'],
     methods: {
      handleClick(){
       this.$emit('addOne',10)
      }
     },
    template: `
       <div @click="handleClick">{{msg}}</div>
    `
  })

// 使用 emits 在发送事件之前做相关校验

  // 使用 `v-model` 来实现父子组件间 数据双向绑定
  // 注意： 子组件中的 modelValue ， update:modelValue 是固定写法
  // v-model 可以写多个
  var app = Vue.createApp({
    data() {
      return {
        obj: 1
      }
    },
    template: `
          <hello-world  v-model="obj"/>
      `,
  })

  // String number Boolean Function 
  app.component('hello-world', {
    props: ['modelValue'],
    methods: {
      handleClick() {
        this.$emit('update:modelValue', this.modelValue + 1)
      }
    },
    template: `
       <div @click="handleClick">{{modelValue}}</div>
    `
  })

                                    # 动态组件

                                    // 普通转换

                                    
var app = Vue.createApp({
    data() {
      return {
        item : 'input-item'
      }
    },
    methods: {
      handleClick(){
          if(this.item === 'input-item'){
              this.item = 'div-item'
          }else{
            this.item = 'input-item'
          }
      }
    },
    template: `
        <input-item v-show="item === 'input-item'"/>
        <div-item v-show="item === 'div-item'"/>
        <button @click="handleClick"> 切换 </button>
      `,
  })

  app.component('input-item', {
    template: `
        <input/>
    `
  })

  app.component('div-item', {
    template: `
        <div>div</div>
    `
  })

                                    动态组件

                                    动态组件 通常与 keep-alive 配合使用，用于保存组件的数据

                                     var app = Vue.createApp({
    data() {
      return {
        item : 'input-item'
      }
    },
    methods: {
      handleClick(){
          if(this.item === 'input-item'){
              this.item = 'div-item'
          }else{
            this.item = 'input-item'
          }
      }
    },
    template: `
        <keep-alive>
          <component :is="item"/>
        </keep-alive>
        <button @click="handleClick"> 切换 </button>
      `,
  })

  app.component('input-item', {
    template: `
        <input/>
    `
  })

  app.component('div-item', {
    template: `
        <div>div</div>
    `
  })

                                    # 异步组件

                                     var app = Vue.createApp({
    data() {
      return {
        item : 'input-item'
      }
    },
    template: `
           <div-item/>
           <async-item/>
      `,
  })

  app.component('div-item', {
    template: `
        <div>div</div>
    `
  })

  app.component('async-item',Vue.defineAsyncComponent(()=>{
      return new Promise((resolve,reject)=>{
         setTimeout(()=>{
             resolve({
              template: `
                  <div>异步组件加载</div>
              `
             })
         },4000)
      })
  }))
