                                                                                                    # v-if v-on

                                                                                                    v-if false 时，不会生成 dom 节点 v-on:click 可以简写成 @click v-bind:title 可以简写成 :title

                                                                                                    
Vue.createApp({
      data() {
        return {
          show: true,
          content: "hello wold",
        };
      },
      methods: {
        clickReverse() {
          this.show = !this.show;
        },
      },

      template: `
      <div>
         <span v-if="show"> {{content}} </span>
         <button v-on:click='clickReverse'>显示/隐藏</button>
        </div>
      `,
    }).mount("#app");

                                                                                                    const app = Vue.createApp({
      // 在实例生成之前自动执行
      beforeCreate() {
        console.log('beforeCreate');
      },
      // 在实例生成之后自动执行
      created() {
        console.log('created');
      },
      // 在组件内容被渲染到页面之前执行
      beforeMount() {
        console.log('beforeMount');
      },
      // 在组件渲染到页面后执行
      mounted() {
        console.log('mounted');
      },

      // 当数据发生变化时会立即执行
      beforeUpdate() {
        console.log('beforeUpdate');
      },
      // 当数据发生变化，页面重选然后，会执行
      updated() {
        console.log('updated');
      },

      beforeUnmount() {
        console.log('beforeUnmount');
      },
      unmounted() {
        console.log('unmoounted');
      },

      beforeDestory(){
        console.log('beforeDestory');
      },
      destoryed(){
        console.log('destoryed');
      },

      data() {
        return {
          content: "",
          list: [],
        };
      },
      methods: {
        add() {
          this.list.push(this.content);
          this.content = "";
        },
      },
      // 属性绑定：v-bind:属性
      template: `
      <div>
        <input v-model="content"/>
        <button 
         v-on:click="add"
         v-bind:title=content
         >添加</button>
         <ul>
             <todo-item v-for="(item,index) of list" 
             v-bind:item=item 
             v-bind:index=index 
             ></todo-item>
          </ul>
      </div>
      `,
    });
