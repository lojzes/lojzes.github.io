                                                     var app = Vue.createApp({
    // conpsAPI vue3 新提供的语法
    setup(props, context) {
      // 
      const { ref, reactive } = Vue;
      const inputValue = ref('')
      const list = reactive([])
      function handleInput(e){
        inputValue.value = e.target.value
      }
      function handleSubmit() {
        list.push(inputValue.value)
         inputValue.value = ''
      }
      return {
        inputValue, list, handleSubmit,handleInput
      }
    },

    template: `
           <div>
            <input :value="inputValue" @input="handleInput"/>
            <button @click="handleSubmit">提交</button>
            </div>

            <ul>
               <li  v-for="(item,index) in list" :key="index">{{item}}</li>
            </ul>
      `,
  })

  // 然后挂载
  var vm = app.mount("#app");

                                                    # 生命周期函数

                                                     var app = Vue.createApp({
    // conpsAPI vue3 新提供的语法
    setup(props, context) {

      // 生命周期函数
      // beforeCreate Created 这两个没有提供，因为setUp 就运行在 这两个生命周期函数内
      // mounted  -> onMounted
      // beforeUpdate -> onBeforeUpdate
      // Updated -> onUpdated
      // beforeUnmount -> onBeforeUnmount
      // Unmounted -> onUnmounted
      // vue3 新的生命周期函数 
      // onRenderTracked  
      // onRenderTriggered
      const { ref,
        onMounted, onBeforeUpdate, onUpdated, onBeforeUnmount, onUnmounted,
        onRenderTracked,onRenderTriggered } = Vue
      const name = ref('lojzes')

      const created = () => {
        console.log('created')
      }

      // 页面挂载完成后执行
      onMounted(() => {
        console.log('onMounted')
      })
      // 页面重新渲染前执行
      onBeforeUpdate(() => {
        console.log('onBeforeUpdate')
      })
      // 页面重新渲染执行
      onUpdated(() => {
        console.log('onUpdated')
      })

      // 收集响应式数据时执行 ，在 onMounted 之前
      onRenderTracked(()=>{
        console.log('onRenderTracked')
      })

      // 每次页面渲染都会执行，在 onBeforeUpdate 之前
      onRenderTriggered(()=>{
        console.log('onRenderTrigged')
      })

      const handleClick = () => {
        name.value = 'lee'
      }

      let isShow = ref(true)
      const handleIfClick = () => {
        isShow.value = !isShow.value
      }
      return {
        name, isShow,
        handleClick, handleIfClick
      }
    },

    template: `
     <div @click="handleClick">
      {{name}}
     </div>

     <div @click="handleIfClick">
      hello : 
      <hello v-if="isShow === true">
     </hello>
     </div>
`,
  })

  app.component('hello', {
    setup(props, context) {

      const { onBeforeUnmount, onUnmounted } = Vue
      // dom元素移除之前执行 举例：子组件的显示与销毁
      onBeforeUnmount(() => {
        console.log('onBeforeUnmount')
      })
      // dom元素移除之后执行 举例：子组件的显示与销毁
      onUnmounted(() => {
        console.log('onUnmounted')
      })
    },
    template: `
     <span>lojzes</span>
    `
  })


  // 然后挂载
  var vm = app.mount("#app");

                                                    # prodive 和 inject

                                                    var app = Vue.createApp({
    // conpsAPI vue3 新提供的语法
    setup(props, context) {
      // provide inject
      const { ref,provide } = Vue
      const name = ref('lojzes')
      // provide 支持多级组件间传值
      //提供 name 和 changeName 方法
      // changeName 提供其他组件进行修改，子组件修改父组件的数据，
      //要在父组件提供的方法进行修改
      provide('name',name)
      provide('changeName',(param)=>{
        name.value = param
      })

      return {
        name
      }
    },

    template: `
     <hello/>
`,
  })

  app.component('hello', {
    setup(props, context) {
      const { inject } = Vue
      const name = inject('name')
      const changeName = inject('changeName')
     
      handleClick = ()=>{
        console.log('handleClick...')
        changeName('lee')
      }
      return {name,handleClick}
    },
    
    template: `
     <div @click="handleClick">{{name}}</div>
    `
  })
  // 然后挂载
  var vm = app.mount("#app");
