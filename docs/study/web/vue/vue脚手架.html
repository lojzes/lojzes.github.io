                                        App.vue

                                        <template>
  <div>
    <input v-model="inputValue" />
    <button class="btn" @click="handleClick">提交</button>
  </div>

  <div>
    <ul>
      <ListItem :list=list></ListItem>
    </ul>
  </div>
</template>

<script>
// 导入
import { ref, reactive } from "vue";
import ListItem from  './components/ListItem';

export default {
  name: "App",
  components:{
    ListItem
  },
  setup() {
    let inputValue = ref("");
    let list = reactive([]);

    const handleClick = () => {
      if (inputValue.value.length !== 0) {
        list.push(inputValue.value);
        inputValue.value = "";
      }
    };

    return {
      list,
      inputValue,
      handleClick,
    };
  },
};
</script>

                                        router/index.js

                                        import { createRouter, createWebHashHistory } from 'vue-router'
import HomeView from '../views/HomeView.vue'

const routes = [
  {
    path: '/',
    name: 'home',
    component: HomeView
  },
  {
    path: '/about',
    name: 'about',
    // route level code-splitting
    // this generates a separate chunk (about.[hash].js) for this route
    // which is lazy-loaded when the route is visited.
    // 懒加载，异步加载
    component: () => import(/* webpackChunkName: "about" */ '../views/AboutView.vue')
  }
]

const router = createRouter({
  history: createWebHashHistory(),
  routes
})

export default router

                                        store.js

                                        import { createStore } from 'vuex'

// vue 数据管理
// 创建了全局唯一的数据仓库 存放了全局的数据
export default createStore({
  state: {
    name : 'lojzes'
  },
  getters: {
  },
  mutations: {
    change(){
      // 第四步 修改数据
      this.state.name = 'lee'
    }
  },
  actions: {
    // 第二步， actions 接收 事件
    change(){
      // 第三步 体积事件
       this.commit('change')
    }
  },
  modules: {
  }
})

                                        注意：如果只涉及同步代码，可以直接在外部调用 this.$store.commit() 即可， 如果涉及异步代码，可以在 actions 写异步代码

                                        传参修改

                                        store.js

                                        import { createStore } from 'vuex'

// vue 数据管理
// 创建了全局唯一的数据仓库 存放了全局的数据
export default createStore({
  state: {
    name: 'lojzes'
  },
  getters: {
  },
  mutations: {

    // 方法1
    // change(state, param) {
    //   // 第四步 修改数据
    //   this.state.name = param
    // }

    // 方法2 
    change: (state, param) => {
      // 第四步 修改数据  不能使用 this.state.name = param
      state.name = param
    }
  },
  actions: {
    // 第二步， actions 接收 事件
    change(store, parma) {
      // 模拟 远程请求接口
      setTimeout(() => {
        // 第三步 体积事件
        store.commit('change', parma)
      }, 2000)
    }
  },
  modules: {
  }
})

                                        import { createStore } from 'vuex'
import axios from "axios";

// vue 数据管理
// 创建了全局唯一的数据仓库 存放了全局的数据
export default createStore({
  state: {
    name: 'lojzes'
  },
  getters: {
  },
  mutations: {
    // 方法1
    // change(state, param) {
    //   // 第四步 修改数据
    //   this.state.name = param
    // }

    // 方法2 
    change(state, param) {
      // 第四步 修改数据  不能使用 this.state.name = param
      state.name = param
      console.log('param', param)
    }
  },
  actions: {
    // 第二步， actions 接收 事件
    change(store) {
      // 第三步 提交事件
      // setTimeout(() => {
      //   // 第三步 提交事件
      //   store.commit('change', parma)
      // }, 2000)

      axios.get("http://localhost:8888/user/get/1").then((response) => {
        store.commit('change', response.data.age)
      });
    }
  },
  modules: {
  }
})
