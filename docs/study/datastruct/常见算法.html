                                        private static void quickSort(int[] a, int start, int end) {
    if (start >= end) {
      return;
    }

    int s = a[start];
    int low = start;
    int high = end;

    while (low < high) {
      while (low < high && s < a[high]) {
        --high;
      }
      a[low] = a[high];
      while (low < high && s > a[low]) {
        low++;
      }
      a[high] = a[low];
    }
    a[low] = s;
    quickSort(a, start, low);
    quickSort(a, low + 1, end);
  }
