                            # 编码、解码

                            打印 unicode、utf-8 编码

                            package main

import "fmt"

func main() {
	// 字节长度
	//fmt.Println(len(s))
	// rune 相当于go 中的char
	var s = "YES你好世界!" // 默认UTF-8  一个中文 占3个字节
	for i, v := range s { // v 是rune unicode、utf-8 编码
		fmt.Printf("(%d %X)\n", i, v)
	}
}

                            # 编码解码

                            var s = "YES你好世界!" // 默认UTF-8  一个中文 占3个字节
	b:= []byte(s)
	
	for len(b) > 0{
		// 解码
		r,size:= utf8.DecodeRune(b)
	
		fmt.Printf("%c  %v\n",r,size)
		b = b[size:]
	}
	// 编码
	r := '好'
	buf := make([]byte, 3)
	utf8.EncodeRune(buf, r)

	fmt.Println(buf)

                            输出

E  1
S  1
你  3
好  3
世  3
界  3
!  1
[229 165 189]

                            # int float string 相互转换

                            // string 转 int
	i,err := strconv.Atoi("10")
	if err == nil{
		fmt.Println(i)
	}
	// int 转 string
	fmt.Println(strconv.Itoa(100))

	// 101 字符串  10进制  32 位
	fmt.Println(strconv.ParseInt("101",10,32))
	// 101.01 字符串转 float
	fmt.Println(strconv.ParseFloat("101.01",64))

	// float 转 string
	fmt.Println(strconv.FormatFloat(101.01,'f',-1,64))

                            # 控制台输入

                            package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	fmt.Println("请输入...")
	 var a string

	for ; ; {
		fmt.Scanf("%s", &a)
		fmt.Printf("输入的内容为 %s\n", a)
	}

	// 读取一行
	fmt.Scanln(&a)
	fmt.Printf("输入的内容为 %s\n", a)
	
	// 使用scanner
	scanner := bufio.NewScanner(os.Stdin)
	for scanner.Scan() {
		s := scanner.Text()
		fmt.Println(s)
	}

	// 使用reader
	r := bufio.NewReader(os.Stdin)
	if r, _, err := r.ReadRune(); err != nil {
		panic(err)
	} else {
		fmt.Println("%v", r)
	}
}
